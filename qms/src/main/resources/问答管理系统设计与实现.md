#### 一、问答管理系统简介

~~~
问答管理系统包含用户管理模块、提问管理、回复管理、评论管理、标签管理、用户管理、日志管理、角色管理、权限管理、数据分析等模块组成根据实际情况抽取核心业务实现即可；该系统服务器端主采用java语言开发、本次选择单体开发模式，该系统本次数据存储主要选择mysql5作为存储数据数据库，Mybatis或MybatisPlus作为数据持久层技术；采用微服务框架Springboot或者SpringCloud作为整合框架技术栈，目的为后期项目拆分微服务版奠定技术基础；采用shiro作为认证登录技术；
~~~

/*==============================================================*/
/* DBMS name:      MySQL 5.0                                    */

drop table if exists answer;

drop table if exists comment;

drop table if exists logs;

drop table if exists permissions;

drop table if exists question;

drop table if exists roles;

drop table if exists tags;

drop table if exists tb_role_per;

drop table if exists tb_user_role;

drop table if exists users;

/*==============================================================*/
/* Table: answer                                                */
/*==============================================================*/
create table answer
(
   aid                  int not null,
   cid                  int,
   uid                  int,
   content              text,
   comment_num          int,
   agree_num            int,
   ctime                varchar(50),
   mtime                varchar(50),
   del                  int,
   primary key (aid)
);

/*==============================================================*/
/* Table: comment                                               */
/*==============================================================*/
create table comment
(
   cid                  int not null,
   uid                  int,
   content              text,
   ctime                varchar(50),
   mtime                varchar(50),
   del                  int,
   primary key (cid)
);

/*==============================================================*/
/* Table: logs                                                  */
/*==============================================================*/
create table logs
(
   operid               int not null,
   opertime             varchar(50),
   opername             varchar(50),
   ip                   varchar(50),
   methods              text,
   ddesc                text,
   primary key (operid)
);

/*==============================================================*/
/* Table: permissions                                           */
/*==============================================================*/
create table permissions
(
   pid                  int not null,
   pname                varchar(50),
   pdesc                varchar(50),
   primary key (pid)
);

/*==============================================================*/
/* Table: question                                              */
/*==============================================================*/
create table question
(
   qid                  int not null,
   uid                  int,
   tid                  int,
   aid                  int,
   title                varchar(50),
   content              text,
   ctime                varchar(50),
   mtime                varchar(50),
   del                  int,
   primary key (qid)
);

/*==============================================================*/
/* Table: roles                                                 */
/*==============================================================*/
create table roles
(
   rid                  int not null,
   rname                varchar(50),
   rdesc                varchar(50),
   primary key (rid)
);

/*==============================================================*/
/* Table: tags                                                  */
/*==============================================================*/
create table tags
(
   tid                  int not null,
   tname                varchar(50),
   ctime                varchar(50),
   mtime                varchar(50),
   primary key (tid)
);

/*==============================================================*/
/* Table: tb_role_per                                           */
/*==============================================================*/
create table tb_role_per
(
   rid                  int not null,
   pid                  int not null,
   primary key (rid, pid)
);

/*==============================================================*/
/* Table: tb_user_role                                          */
/*==============================================================*/
create table tb_user_role
(
   uid                  int not null,
   rid                  int not null,
   primary key (uid, rid)
);

/*==============================================================*/
/* Table: users                                                 */
/*==============================================================*/
create table users
(
   uid                  int not null,
   uname                varchar(50),
   nickname             varchar(50),
   password             varchar(50),
   sex                  varchar(50),
   avatar               text,
   remark               varchar(50),
   ctime                varchar(50),
   mtime                varchar(50),
   del                  int,
   primary key (uid)
);
